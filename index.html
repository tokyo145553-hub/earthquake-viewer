function normalizeCityName(raw){ if (!raw) return "不明"; raw = raw.replace( /[０-９Ａ-Ｚａ-ｚ]/g, s => String.fromCharCode( s.charCodeAt(0) - 65248 ) ); raw = raw.replace(/\s+/g, ""); raw = raw.replace(/（.*?）/g, ""); raw = raw.replace(/[-ー－]+$/g, ""); const specialCities = [ "田村市","市川市","市原市", "野々市市","四日市市","町田市", "十日町市","大町市","山村市", "村山市","東村山市","武蔵村山市", "羽村市","大村市","余市町", "市貝町","上市町","市川三郷町", "市川町","下市町","大町町", "村田町","玉村町","中村区" ]; if(specialCities.includes(raw)){ return raw; } const posCity = raw.indexOf("市"); if(posCity > 0){ return raw.substring(0, posCity + 1); } const chars = [ "区", "町", "村" ]; let cutPos = -1; for(const c of chars){ const pos = raw.indexOf(c); if( pos > 0 && (cutPos === -1 || pos < cutPos) ){ cutPos = pos; } } if(cutPos > 0){ return raw.substring(0, cutPos + 1); } return raw; }
